5.1. TRIGGER вЂ“ PRICE CHANGE
CREATE OR REPLACE FUNCTION log_price_change()
RETURNS TRIGGER AS $$
BEGIN
  INSERT INTO pricing_logs(old_price, new_price)
  VALUES (OLD.base_price, NEW.base_price);
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;
5.2. RECURSIVE CTE (BOOKING DATE RANGE)
WITH RECURSIVE dates AS (
  SELECT check_in AS d
  UNION ALL
  SELECT d + INTERVAL '1 day'
  FROM dates
